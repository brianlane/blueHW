Trading Event ContractsWhich Webull platforms support trading Event Contracts?Event Contracts are available on both the Webull mobile app and the desktop platform. On desktop, availability is currently limited to index and crypto hourly contracts, while the full Event Contracts offering is accessible through the mobile app. We look forward to continuing to develop and enhance the product, incorporating invaluable input from our valued clients!How do I trade Event Contracts?To begin trading Event Contracts, you’ll need to submit an application through your existing Webull brokerage account. The process is quick and straightforward to help you get started smoothly.Event Contract accounts are available only to U.S. residents.How do I open a position in an Event Contract?  To open a position in an Event Contract on the Webull mobile app, navigate to Markets> ‘Events.’ From here, select an Event Contract to trade. To open a position, choose “Yes” or “No” for a particular event and confirm the order submission.  How do I close a position in an Event Contract? Once a position has been opened through purchasing “Yes” or “No” contracts, there are two ways to exit the position. A client can close the position by selling to close the contract and realizing any profit or loss associated with the prices at which they purchased and sold the contract. A client may also hold the contract through expiration, at which point the market will either resolve in their favor and pay out $1 per contract or it will not resolve in their favor and pay out $0. What are the trading hours for Event Contracts? Trading hours vary by contract type and market. import requests
import datetime
import base64
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.asymmetric import padding
from cryptography.hazmat.primitives import serialization
from cryptography.hazmat.backends import default_backend

# ================== CONFIG ==================
DEMO = True  # Use sandbox (recommended for application)
BASE_URL = "https://demo-api.kalshi.co" if DEMO else "https://api.elections.kalshi.com"

# YOUR CREDENTIALS (get these from kalshi.com/account/profile → API Keys)
KEY_ID = "your-key-id-here"                    # e.g. a952bcbe-ec3b-4b5b-b8f9-11dae589608c
PRIVATE_KEY_PATH = "kalshi_private_key.pem"    # the .pem file you downloaded

# ===========================================

def load_private_key(file_path):
    with open(file_path, "rb") as f:
        return serialization.load_pem_private_key(f.read(), password=None, backend=default_backend())

def sign_request(private_key, timestamp_ms: str, method: str, path: str) -> str:
    msg = f"{timestamp_ms}{method}{path}".encode("utf-8")
    signature = private_key.sign(
        msg,
        padding.PSS(
            mgf=padding.MGF1(hashes.SHA256()),
            salt_length=padding.PSS.DIGEST_LENGTH,
        ),
        hashes.SHA256(),
    )
    return base64.b64encode(signature).decode("utf-8")

def get_headers(method: str, path: str):
    timestamp_ms = str(int(datetime.datetime.now().timestamp() * 1000))
    private_key = load_private_key(PRIVATE_KEY_PATH)
    signature = sign_request(private_key, timestamp_ms, method, path)
    return {
        "KALSHI-ACCESS-KEY": KEY_ID,
        "KALSHI-ACCESS-TIMESTAMP": timestamp_ms,
        "KALSHI-ACCESS-SIGNATURE": signature,
        "Content-Type": "application/json",
    }

# ================== STEP 1: Market Data (NYC Weather) ==================
print("1. Querying NYC weather market data...")
series_url = f"{BASE_URL}/trade-api/v2/series/KXHIGHNY"
series_resp = requests.get(series_url).json()
print(f"Series: {series_resp['series']['title']}")

markets_url = f"{BASE_URL}/trade-api/v2/markets?series_ticker=KXHIGHNY&status=open"
markets_resp = requests.get(markets_url).json()
ticker = markets_resp["markets"][0]["ticker"]   # first open market for today
print(f"Using market ticker: {ticker} ({markets_resp['markets'][0]['title']})")

# ================== STEP 2: Orderbook ==================
print("\n2. Querying orderbook...")
orderbook_url = f"{BASE_URL}/trade-api/v2/markets/{ticker}/orderbook"
orderbook_resp = requests.get(orderbook_url).json()
print(f"YES bids top 3: {orderbook_resp['orderbook_fp']['yes_dollars'][:3]}")
print(f"NO bids top 3: {orderbook_resp['orderbook_fp']['no_dollars'][:3]}")

# ================== STEP 3: Place + Cancel Order (1 unit) ==================
print("\n3. Placing and canceling a test order (demo only)...")

# Place order (buy 1 unit YES at $0.50)
path = "/trade-api/v2/portfolio/orders"
place_headers = get_headers("POST", path)
place_body = {
    "ticker": ticker,
    "action": "buy",
    "side": "yes",
    "count_fp": 1,               # 1 unit
    "yes_price_dollars": 0.50    # safe demo price
}

place_resp = requests.post(f"{BASE_URL}{path}", headers=place_headers, json=place_body)
order_id = place_resp.json().get("order_id") if place_resp.status_code == 200 else None
print(f"Order placed → ID: {order_id}")

# Cancel the order
if order_id:
    cancel_path = f"/trade-api/v2/portfolio/orders/{order_id}"
    cancel_headers = get_headers("DELETE", cancel_path)
    cancel_resp = requests.delete(f"{BASE_URL}{cancel_path}", headers=cancel_headers)
    print(f"Order canceled → Status: {cancel_resp.status_code} ({cancel_resp.text[:200]})")
else:
    print("Order placement failed — check your key/permissions.")

print("\n✅ All steps complete! Upload this file as proof of advanced API capability.")

def create_signature(method: str, path: str, timestamp: str) -> str:
    msg = timestamp + method + path
    signature = private_key.sign(

        padding.PSS(
            mgf=padding.MGF1(hashes.SHA256()),
            salt_length=padding.PSS.DIGEST_LENGTH

        hashes.SHA256()
    )
